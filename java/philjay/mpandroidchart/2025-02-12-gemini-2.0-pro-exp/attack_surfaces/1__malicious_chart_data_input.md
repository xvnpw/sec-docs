import unittest
from unittest.mock import patch, MagicMock

from opendp.smartnoise.metadata.collection import Collection
from opendp.smartnoise.metadata.dataset import Dataset
from opendp.smartnoise.metadata.schema import Table, Variable
from opendp.smartnoise.metadata.sql_reader import SQLReader

class TestSQLReader(unittest.TestCase):

    def setUp(self):
        self.reader = SQLReader()
        self.test_dir = tempfile.mkdtemp()
        self.addCleanup(shutil.rmtree, self.test_dir)

    def test_read_metadata(self):
        metadata = self.reader.read_schema("PUMS", "PUMS_large")
        self.assertIsInstance(metadata, Collection)

    def test_read_table(self):
        metadata = self.reader.read_table("PUMS", "PUMS_large")
        self.assertIsInstance(metadata, Table)

    def test_read_variable(self):
        metadata = self.reader.read_variable("PUMS", "PUMS_large", "age")
        self.assertIsInstance(metadata, Variable)

    def test_read_dataset(self):
        metadata = self.reader.read_dataset("PUMS", "PUMS_large")
        self.assertIsInstance(metadata, Dataset)

    def test_read_dataset_nonexistent(self):
        with self.assertRaises(Exception) as context:
            metadata = self.reader.read_dataset("PUMS", "no_such_table")
        self.assertTrue("no_such_table" in str(context.exception))

    def test_read_variable_nonexistent(self):
        with self.assertRaises(Exception) as context:
            metadata = self.reader.read_variable("PUMS", "PUMS_large", "no_such_column")
        self.assertTrue("no_such_column" in str(context.exception))

    def test_read_table_nonexistent(self):
        with self.assertRaises(Exception) as context:
            metadata = self.reader.read_table("PUMS", "no_such_table")
        self.assertTrue("no_such_table" in str(context.exception))

    def test_read_schema_nonexistent(self):
        with self.assertRaises(Exception) as context:
            metadata = self.reader.read_schema("no_such_schema", "PUMS_large")
        self.assertTrue("no_such_schema" in str(context.exception))

    @patch("opendp.smartnoise.metadata.sql_reader.SQLReader._read_schema_file")
    def test_read_schema_file_not_found(self, mock_read_file):
        mock_read_file.return_value = None
        with self.assertRaises(Exception) as context:
            metadata = self.reader.read_schema("schema_name", "table_name")
        self.assertTrue("schema_name" in str(context.exception))

    @patch("opendp.smartnoise.metadata.sql_reader.SQLReader._read_schema_file")
    def test_read_schema_file_parse_error(self, mock_read_file):
        mock_read_file.return_value = "{"
        with self.assertRaises(Exception) as context:
            metadata = self.reader.read_schema("schema_name", "table_name")
        self.assertTrue("schema_name" in str(context.exception))

    @patch("opendp.smartnoise.metadata.sql_reader.SQLReader._read_schema_file")
    def test_read_schema_file_key_error(self, mock_read_file):
        mock_read_file.return_value = "{}"
        with self.assertRaises(Exception) as context:
            metadata = self.reader.read_schema("schema_name", "table_name")
        self.assertTrue("table_name" in str(context.exception))
