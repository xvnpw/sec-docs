#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <cctype>

using namespace std;

// Function to check if a character is a vowel
bool isVowel(char c) {
    c = tolower(c);
    return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
}

int main() {
    string s;
    cin >> s;

    vector<int> vowelIndices;
    for (int i = 0; i < s.length(); ++i) {
        if (isVowel(s[i])) {
            vowelIndices.push_back(i);
        }
    }

    // If there are less than 2 vowels, no operation is needed
    if (vowelIndices.size() < 2) {
        cout << s << endl;
        return 0;
    }

    // Swap the first and last vowels
    swap(s[vowelIndices[0]], s[vowelIndices.back()]);

    cout << s << endl;

    return 0;
}
